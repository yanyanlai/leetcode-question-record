# 704.二分查找

```c++
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int l =0;
        int r =nums.size() - 1;
        while(r>=l){
            int m = (l+r)/2;
            if(target > nums[m]){
                l = m + 1;
            }
            else if(target < nums[m]){
                r = m - 1;
            }
            else{
                return m;
            }
        }
        return -1;
    }
};
```
